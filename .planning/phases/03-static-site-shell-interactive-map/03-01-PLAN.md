PMTiles + basemap recipe (RESEARCH Pattern 1, verified):
```typescript
import * as maplibregl from "maplibre-gl";               // NAMESPACE import — default import breaks Vite dev on 5.x
import "maplibre-gl/dist/maplibre-gl.css";
import { Protocol } from "pmtiles";
import { layers, namedFlavor } from "@protomaps/basemaps";
const protocol = new Protocol();
maplibregl.addProtocol("pmtiles", protocol.tile);
const BASE = import.meta.env.BASE_URL;                    // "/betravedur/" in prod, "/" in dev — written verbatim
const pmtilesUrl = `pmtiles://${location.origin}${BASE}iceland.pmtiles`;
// style: { version:8, glyphs:"https://protomaps.github.io/basemaps-assets/fonts/{fontstack}/{range}.pbf",

//   sources:{ protomaps:{ type:"vector", url:pmtilesUrl, attribution:"© OpenStreetMap · Protomaps" } },
//   layers: layers("protomaps", namedFlavor("grayscale"), { lang:"is" }) }
```
