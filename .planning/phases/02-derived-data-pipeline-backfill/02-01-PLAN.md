From packages/domain/src/types.ts (re-exported via @betravedur/domain):
```typescript
export interface DailyObservation {
  station: number;
  date: string;      // "YYYY-MM-DD"
  doy: number;       // leap-folded day-of-year 1..365 (Feb 29 excluded)
  t: number | null; tx: number | null; tn: number | null;   // temps °C
  f: number | null; fx: number | null; fg: number | null;   // wind m/s
  dv: number | null; // wind dir degrees 0-360; null on SYNOP
  r: number | null;  // precip mm; null on AWS; missing NEVER coerced to 0
}
export type StationType = "sj" | "sk" | "ur" | "vf"; // sj=AWS, sk=SYNOP
export interface WindowSpec { startDoy: number; endDoy: number; } // wraps when endDoy < startDoy

export function groupBySeasonYear(rows: DailyObservation[], spec: WindowSpec): Map<number, DailyObservation[]>;
// Dec head owns the year (WR-03). For a wrapping window (endDoy<startDoy):
//   doy>=startDoy -> season = calendarYear; doy<=endDoy -> season = calendarYear-1.
export function expandWindow(spec: WindowSpec): Set<number>;   // leap-folded doy set
export function leapFoldedDoy(date: string): number | null;    // 1..365, null for Feb 29 / bad date
```
